###### // 重组 = 从数组和对象中提取值、
###### // 然后以方便的方式将它们赋值给变量
###### // [] = 执行数组重组
###### // {} = 对对象进行重组

---

```js
// --------- EXAMPLE 1 --------- 
// 交换两个变量的值
let a = 1; 
let b = 2; 
[a, b] = [b, a]; 
console.log(a); //2
console.log(b); //1

// --------- EXAMPLE 2 --------- 
// 交换数组中的 2 个元素
const colors = ['red', 'green', 'blue', 'black', 'white']; 
[colors[0], colors[4]] = [colors[4], colors[0]] 
console.log(colors);	//['white', 'green', 'blue', 'black', 'red']
// --------- EXAMPLE 3 --------- 
// 将数组元素分配给变量 
const [firstColor, secondColor, thirdColor, ...extraColors] = colors; console.log(firstColor); 	//white
console.log(secondColor); 	//green
console.log(thirdColor); 	//blue
console.log(extraColors);	//['black', 'red']

// --------- EXAMPLE 4 --------- 
// 从对象中提取值 
const person1 = { 
	firstName: 'Spongebob', 
	lastName: 'Squarepants', 
	age: 30, 
	job: "Fry cook", 
}; 
const person2 = { 
	firstName: 'Patrick', 
	lastName: 'Star', 
	age: 34 
}; 
const {firstName, lastName, age, job="Unemployed"} = person2; console.log(firstName); 
console.log(lastName); 
console.log(age); 
console.log(job);
```
