using System;

namespace MyFirstProgram
{
    class Program
    {
        static void Main(string[] args)
        {
            Random random = new Random();
            bool playAgain = true;
            String player;
            String computer;
            String answer;

            while (playAgain)
            {
                player = "";
                computer = "";
                answer = "";

                while (player != "石头" && player != "布" && player != "剪刀") {
                    Console.Write("Enter 石头, 布, or 剪刀: ");
                    player = Console.ReadLine();
                    player = player.ToUpper();
                }

                
                switch (random.Next(1, 4))
                {
                    case 1:
                        computer = "石头";
                        break;
                    case 2:
                        computer = "布";
                        break;
                    case 3:
                        computer = "剪刀";
                        break;
                }

                Console.WriteLine("Player: " + player);
                Console.WriteLine("Computer: " + computer);

                switch (player)
                {
                    case "石头":
                        if (computer == "石头")
                        {
                            Console.WriteLine("It's a draw!");
                        }
                        else if (computer == "布")
                        {
                            Console.WriteLine("You lose!");
                        }
                        else
                        {
                            Console.WriteLine("You win!");
                        }
                        break;
                    case "布":
                        if (computer == "石头")
                        {
                            Console.WriteLine("You win!");
                        }
                        else if (computer == "布")
                        {
                            Console.WriteLine("It's a draw!");
                        }
                        else
                        {
                            Console.WriteLine("You lose!");
                        }
                        break;
                    case "剪刀":
                        if (computer == "石头")
                        {
                            Console.WriteLine("You lose!");
                        }
                        else if (computer == "布")
                        {
                            Console.WriteLine("You win!");
                        }
                        else
                        {
                            Console.WriteLine("It's a draw!");
                        }
                        break;
                }

                Console.Write("Would you like to play again (Y/N): ");
                answer = Console.ReadLine();
                answer = answer.ToUpper();

                if (answer == "Y")
                {
                    playAgain = true;
                }
                else
                {
                    playAgain = false;
                }
                
            }

            Console.WriteLine("Thanks for playing!");

            Console.ReadKey();
        }
    }
}
