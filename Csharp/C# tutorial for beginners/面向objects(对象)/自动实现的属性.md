#### // 自动实现的属性 = 属性中不需要额外逻辑时的快捷方式 
#### // 您不必为属性定义字段， 
#### // 你只需要写 get; and/or set; 内部

---

```c#
using System;

namespace MyFirstProgram
{
    class Program 
    {
        static void Main(string[] args)
        {
            Car car = new Car("Porsche");

            Console.WriteLine(car.Model);

            Console.ReadKey();
        }
    }

    class Car
    {
        public String Model {get; set;}

        public Car(String model)
        {
            this.Model = model;
        }
    }
}
```
