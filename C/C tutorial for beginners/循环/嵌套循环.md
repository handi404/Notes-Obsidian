//嵌套循环=一个循环嵌套在另一个循环中

---

```c
#include <stdio.h>

int main()
{
   int rows;
   int columns;
   char symbol;

   printf("\nEnter # of rows: ");
   scanf("%d", &rows);

   printf("Enter # of columns: ");
   scanf("%d", &columns);

   scanf("%c"); //从缓冲区中清除\n

   printf("Enter a symbol to use: ");
   scanf("%c", &symbol);

   for(int i = 1; i <= rows; i++)
   {
      for(int j = 1; j <= columns; j++)
      {
         printf("%d", j);
      }
      printf("\n");
   }

   return 0;
}
```
