// struct = 相关成员的集合（“变量”）
// 它们可以是不同的数据类型
// 在内存块中以一个名称列出
// 与其他语言中的类非常相似（但没有方法）

---

```c
#include <stdio.h>
#include <string.h>

struct Player
{
   char name[12];
   int score;
};

int main()
{
   struct Player player1;
   struct Player player2;

   strcpy(player1.name, "Bro");
   player1.score = 4;

   strcpy(player2.name, "Bra");
   player2.score = 5;

   printf("%s\n", player1.name);
   printf("%d\n", player1.score);

   printf("%s\n", player2.name);
   printf("%d\n", player2.score);

   return 0;
}
```
