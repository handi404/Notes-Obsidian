
```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main()
{
   const int MIN = 1;
   const int MAX = 100;
   int guess = 0;
   int guesses = 0;
   int answer = 0;

   //使用当前时间作为种子
   srand(time(0));
   
   //生成一个介于MIN和MAX之间的随机数
   answer = (rand() % MAX) + MIN;

   do{
      printf("Enter a guess: ");
      scanf("%d", &guess);
      if(guess > answer)
      {
         printf("Too high!\n");
      }
      else if(guess < answer)
      {
         printf("Too low!\n");
      }
      else{
         printf("CORRECT!\n");
      }
      guesses++;
   }while(guess != answer);

   printf("**********************\n");
   printf("answer: %d\n", answer);
   printf("guesses: %d\n", guesses);
   printf("**********************");
 
   return 0;
}
```
