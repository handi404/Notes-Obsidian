
```c
#include <stdio.h>
#include <stdbool.h>

int main(){

    char a = 'C';                  // 单个字符 %c
    char b[] = "Bro";            // 字符数组 %s 

    float c = 3.141592;     // 4 字节（32 位精度）6 - 7 位 %f
    double d = 3.141592653589793; // 8 字节（64 位精度）15 - 16 位 %lf

    bool e = true;                // 1 字节（真或假）%d
 
    char f = 120;                // 1 字节（-128 到 +127）%d 或 %c
    unsigned char g = 255;       // 1 字节（0 到 +255）%d 或 %c

    short h = 32767;                  // 2 个字节（−32,768 到 +32,767）%d
    unsigned short i = 65535;  // 2 个字节（0 到 +65,535）%d

    int j = 2147483647;     // 4 个字节（-2,147,483,648 到 +2,147,483,647）%d
    unsigned int k = 4294967295;  // 4 个字节（0 到 +4,294,967,295）%u

    long long int l = 9223372036854775807;//8 个字节（-9 quintillion 到 +9 quintillion） %lld
    unsigned long long int m = 18446744073709551615U; //8 个字节（0 到 +18 quintillion） %llu

    printf("%c\n", a);  // 字符char
    printf("%s\n", b);  // 字符数组character array
    printf("%f\n", c);  // 浮点数float
    printf("%lf\n", d); // 双精度浮点数double
    printf("%d\n", e);  // 布尔值bool
    printf("%d\n", f);  // 字符作为数值char as numeric value
    printf("%d\n", g);  // 无符号字符作为数值unsigned char as numeric value
    printf("%d\n", h);  // 短的short
    printf("%d\n", i);  // 无符号短unsigned short
    printf("%d\n", j);  // 整数int
    printf("%u\n", k);  // 无符号整型unsigned int
    printf("%lld\n", l);  // 长整型long long int
    printf("%llu\n", m);  // 无符号长整型unsigned long long int

    return 0;
}
```
