// 格式说明符 % = 定义并格式化要显示的数据类型

// %c = 字符
// %s = 字符串（字符数组）
// %f = 浮点数
// %lf = 双精度
// %d = 整数

// %. 1 = 小数精度
// %1 = 最小字段宽度
// %- = 左对齐

---

```c
#include <stdio.h>

int main(){
    float item1 = 5.75;
    float item2 = 10.00;
    float item3 = 100.99;

    printf("Item 1: $%8.2f\n", item1);
    printf("Item 2: $%8.2f\n", item2);
    printf("Item 3: $%8.2f\n", item3);
    
    return 0;
}
```
