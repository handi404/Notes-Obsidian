#### // 类型转换 = 将一种数据类型的值转换为另一种数据类型 
#### // 隐式 = 自动 
#### // 显式 = 在值前添加新数据类型 (int)

---

```c++
#include<iostream>
 
int main()
{
	//char x = 100; //隐式
	//int number = 3.134; 
	//std::cout << x <<'\n'; //d
    int correct = 8;
    int questions = 10;
    double score = correct/(double)questions * 100;

    std::cout << score << "%";

    return 0;
}
```
