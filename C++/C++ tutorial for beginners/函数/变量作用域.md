[[变量作用域运算符]]

```c++
#include <iostream>

int myNum = 3; //全局

void printNum();

int main()
{
    int myNum = 1; //局部
    printNum();
    std::cout << "main: " << myNum << '\n'; //局部
    //std::cout << ::myNum << '\n'; //全局

    return 0;
}
void printNum(){
    int myNum = 2; //局部
    std::cout << "printNum: "<< myNum << '\n'; //局部
    //std::cout << ::myNum << '\n'; //全局
}
```
