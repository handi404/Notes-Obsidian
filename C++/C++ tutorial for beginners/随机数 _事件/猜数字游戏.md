#include <iostream>
#include <cmath>
#include <random>

int main(){
    int num = 0;
    int guess = 0;
    int tries = 0;

    std::random_device rd;
    std::mt19937 gen(rd());
    std::uniform_int_distribution<> dis(1, 100);
    num = dis(gen);
    std::cout << "**** NUMBER GUESSING GAME ******\n";
    do
    {
        /* code */
        std::cout << "Enter a guess between (1-100): ";
        std::cin >> guess;
        tries++;
        if (guess > num)
        {
            /* code */
            std::cout << "Too high" << std::endl;
        }else if (guess < num)
        {
            /* code */
            std::cout << "Too low" << std::endl;
        }else{
            std::cout << "CORRECT! # of tries: " << tries << '\n';
            break;
        }

    } while (guess != num);
    
    std::cout << "**********************************";
    return 0;
}
