
```C++
#include <iostream>

void swap(std::string &x, std::string &y);
void noswap(std::string x, std::string y);
int main()
{
	std::string x = "Kool-Aid";
	std::string y = "Water";

	std::cout << "X: " << &x << '\n';	
 	std::cout << "Y: " << &y << '\n';
	noswap(x, y);
	std::cout << "X: " << x << '\n';
 	std::cout << "Y: " << y << '\n';
	
	swap(x, y);
	std::cout << "X: " << x << '\n';
	std::cout << "Y: " << y << '\n';

    return 0;
}
void swap(std::string &x, std::string &y){
	std::string temp;
	temp = x;
	x = y;
	y = temp;
}
void noswap(std::string x, std::string y){
	//这里x y是副本地址不同，无法将main里的x y值进行交换
	std::cout << "X: " << &x << '\n';	
 	std::cout << "Y: " << &y << '\n';
	std::string temp;
 	temp = x;
 	x = y;
 	y = temp;
}
```
